82  void AllocCpuPtr(int id, size_t count) {
83  cpu_pinned_copy_ = op_kernel_->AllocateBufferOnCPUPinned<T>(id, count);
84  if (cpu_pinned_copy_ == nullptr)
85  throw std::runtime_error("alloc failed");
86  count_ = count;
87  }
88 
89  Status CopyToGpu() {
90  if (cpu_pinned_copy_) {
91  gpu_copy_ = op_kernel_->GetScratchBuffer<T>(count_);
92  CUDA_RETURN_IF_ERROR(cudaMemcpyAsync(gpu_copy_.get(), cpu_pinned_copy_.get(), count_ * sizeof(T), cudaMemcpyHostToDevice));
93  op_kernel_->AddDeferredReleaseCPUPtr(cpu_pinned_copy_.release());
94  }
95  return Status::OK();
96  }

163 inline bool CalculateFdmStrides(gsl::span<fast_divmod> p, const std::vector<int64_t>& dims) {
164  int stride = 1;
165  if (dims.empty() || p.size() < gsl::narrow_cast<ptrdiff_t>(dims.size()))
166  return false;
167  std::ptrdiff_t rank = p.size();
168  for (std::ptrdiff_t i = 0; i < rank; i++) {
169  p[rank - 1 - i] = fast_divmod(stride);
170  if (static_cast<size_t>(i) < dims.size() - 1) {
171  stride *= static_cast<int>(dims[dims.size() - 1 - i]);
172  }
173  }
174  return true;
175 }
