12 // Verify and convert unknown dim during reshape
13 class ReshapeHelper {
14  public:
15  ReshapeHelper(const TensorShape& input_shape, std::vector<int64_t>& requested_shape) {
16  auto nDims = requested_shape.size();
17  int64_t unknown_dim = -1;
18  int64_t size = 1;
19  for (size_t i = 0; i < nDims; ++i) {
20  ONNXRUNTIME_ENFORCE(requested_shape[i] >= -1, "A dimension cannot be less than -1.");
21  if (requested_shape[i] == -1) {
22  ONNXRUNTIME_ENFORCE(unknown_dim == -1, "At most one dimension can be -1.");
23  unknown_dim = i;
24  } else {
25  if (requested_shape[i] == 0) {
26  ONNXRUNTIME_ENFORCE(i < input_shape.NumDimensions(),
27  "The dimension with value zero exceeds"
28  " the dimension size of the input tensor.");
29  requested_shape[i] = input_shape[i];
30  }
31  size *= requested_shape[i];
32  }
33  }
34 
35  if (unknown_dim != -1) {
36  // calculate unknown dimension
37  ONNXRUNTIME_ENFORCE((input_shape.Size() % size) == 0,
38  "The input tensor cannot be reshaped to the requested shape. Input shape:", input_shape);
39  requested_shape[unknown_dim] = input_shape.Size() / size;
40  } else {
41  // check if the output shape is valid.
42  ONNXRUNTIME_ENFORCE(gsl::narrow_cast<int64_t>(input_shape.Size()) == size,
43  "The input tensor cannot be reshaped to the requested shape. Input shape:", input_shape);
44  }
45  }
46 };
