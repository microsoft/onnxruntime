17  if (status.IsOK()) {

21  // Check that perm_ is a valid permutation of [0,rank-1]
22  for (auto i : perm_) {
23  if ((i < 0) || (i >= gsl::narrow<int64_t>(rank)))
24  ONNXRUNTIME_THROW("Attribute perm of Transpose has an invalid value. Value ", i, " is outside range.");
25  if (seen[i])
26  ONNXRUNTIME_THROW("Attribute perm of Transpose has an invalid value. Value ", i, " is repeated.");

33  std::vector<int64_t>& default_perm, const std::vector<int64_t>*& p_perm) const {
34  size_t rank = X.Shape().NumDimensions();
35  const auto& input_dims = X.Shape().GetDims();
36 
37  // Determine permutation to use:
38  // If no permutation was specified in the attributes, the default is [rank-1, ..., 0]
39  default_perm.resize(rank);
40 
41  if (perm_specified_)
42  p_perm = &perm_;
43  else {
44  for (int i = 0; i < rank; ++i)
45  default_perm[i] = rank - i - 1;
46  p_perm = &default_perm;
47  }
48 
49  // Determine shape of output, as well as stride to be used:
50  // stride[i] indicates the stride for the input-tensor dimension corresponding
51  // to the i-th dimension of the output
52 
53  output_dims.resize(rank);
54  for (int i = 0; i < rank; i++) {
55  size_t inpdim = (*p_perm)[i];
56  output_dims[i] = input_dims[inpdim];
57  }
58  }
