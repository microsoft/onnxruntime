101 // reverse an LSTM or GRU sequence which has shape [seq_length, batch_size, hidden_size]
102 // and output to shape [seq_length, num_directions, batch_size, hidden_size]
103 template <typename T>
104 void ReverseSequence(gsl::span<const T> inputs,
105  gsl::span<T> inputs_reverse,
106  gsl::span<const int> sequence_lengths,
107  const int max_sequence_length,
108  const int batch_size,
109  const int input_size,
110  const int num_directions) {
111  for (int i = 0; i < batch_size; i++) {
112  int seq_len = sequence_lengths[i];
113 
114  if (seq_len == 0)
115  continue;
116 
117  // Parallel execute the loop.
118  #pragma omp for
119  for (int j = 0; j < seq_len; j++) {
120  gsl::span<const T> src = inputs.subspan(j * batch_size * input_size + i * input_size, input_size);
121  gsl::span<T> dest = inputs_reverse.subspan(num_directions * (seq_len - j - 1) * batch_size * input_size + i * input_size, input_size);
122 
123  // Use gsl::copy instead of std::copy() to allow compiler to optimize the code
124  gsl::copy(src, dest);
125  }
126 
127  #pragma omp for
128  for (int j = seq_len; j < max_sequence_length; j++) {
129  gsl::span<const T> src = inputs.subspan(j * batch_size * input_size + i * input_size, input_size);
130  gsl::span<T> dest = inputs_reverse.subspan(num_directions * j * batch_size * input_size + i * input_size, input_size);
131 
132  // Use gsl::copy instead of std::copy() to allow compiler to optimize the code
133  gsl::copy(src, dest);
134  }
135  }
136 }

138 // A has size M x K, B has size N x K (transposed), and C has size M x N
139 // We check that A, B and C are large enough before calling the lower level GEMM implementation
140 template <typename TSpanAIter, typename TSpanBIter, typename TSpanCIter>
141 void ComputeGemm(const int M,
142  const int N,
143  const int K,
144  const float alpha,
145  TSpanAIter A,
146  TSpanAIter A_end,
147  const int lda,
148  TSpanBIter B,
149  TSpanBIter B_end,
150  const int ldb,
151  const float beta,
152  TSpanCIter C,
153  TSpanCIter C_end,
154  const int ldc) {
155  // validate all the inputs
156  // need to use the lda/ldb/ldc strides which should be >= the columns for the span
157  ONNXRUNTIME_ENFORCE(lda >= K && ldb >= K && ldc >= N);
158  ONNXRUNTIME_ENFORCE(A + (M * lda - (lda - K)) <= A_end);
159  ONNXRUNTIME_ENFORCE(B + (N * ldb - (ldb - K)) <= B_end);
160  ONNXRUNTIME_ENFORCE(C + (M * ldc - (ldc - N)) <= C_end);
161 
162  ::onnxruntime::math::GemmEx<float, CPUMathUtil>(
163  CblasNoTrans, CblasTrans,
164  M, N, K, alpha,
165  &*A, lda,
166  &*B, ldb, beta,
167  &*C, ldc, &CPUMathUtil::Instance());
168 }

206 template <typename TLambda>
207 void ExecuteLambdaInParallel(const std::string& name, TLambda lambda, int max, int step,
208  TaskThreadPool& ttp, const ::onnxruntime::logging::Logger& logger) {
209  // #define NOTHREADS to execute the lambdas directly and in order if you need to do that to debug
210 
211 #ifdef NOTHREADS
212  ONNXRUNTIME_UNUSED_PARAMETER(ttp);
213  ONNXRUNTIME_UNUSED_PARAMETER(logger);
214 
215  for (int i = 0; i < max; i += step) {
216  (void)name;
217  std::bind(lambda, i)();
218  }
219 #else
220  std::vector<std::future<void> > task_results{};
221  task_results.reserve(static_cast<size_t>(std::ceil(max / step)));
222 
223  for (int i = 0; i < max; i += step) {
224  std::packaged_task<void()> task{std::bind(lambda, i)};
225  task_results.emplace_back(task.get_future());
226  ttp.RunTask(std::move(task));
227  }
228 
229  try {
230  // wait for all and propagate any exceptions
231  for (auto& future : task_results)
232  future.get();
233  } catch (const std::exception& ex) {
234  LOGS(logger, ERROR) << name << " - exception running tasks: " << ex.what();
235  throw;
236  }
237 #endif
238 }
