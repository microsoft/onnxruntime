13 /*
14 Given a tensor input, this operation inserts a dimension of 1 at the dimension index axis
15 of X's shape. The dimension index axis starts at zero; if you specify a negative number
16 of axis, it starts backward from the end.
17 */
18 
19 class ExpandDims final : public OpKernel {
20  public:
21  explicit ExpandDims(const OpKernelInfo& info) : OpKernel(info) {}
22 
23  Status Compute(OpKernelContext* context) const override {
24  const Tensor* axis_tensor = context->Input<Tensor>(1);
25  if (axis_tensor == nullptr) return Status(common::ONNXRUNTIME, common::FAIL, "input count mismatch");
26 
27  ONNXRUNTIME_ENFORCE(axis_tensor->Shape().IsScalar(), "An axis tensor must be a scalar tensor.");
28  const int64_t axis = static_cast<int64_t>(axis_tensor->template Data<int32_t>()[0]);
29  const Tensor* X = context->Input<Tensor>(0);
30  if (X == nullptr) return Status(common::ONNXRUNTIME, common::FAIL, "input count mismatch");
31  const TensorShape& X_shape = X->Shape();
32 
33  std::vector<int64_t> expanded_shape(X_shape.GetDims());
34  int64_t X_NumDims = X_shape.Size();
35  ONNXRUNTIME_ENFORCE(axis <= X_NumDims && axis >= -X_NumDims,
36  "Axis must be within range [", -X_NumDims, ", ", X_NumDims, "].", " Axis is ", axis);
37  if (axis >= 0) {
38  expanded_shape.insert(expanded_shape.begin() + axis, 1);
39  } else {
40  expanded_shape.insert(expanded_shape.end() + axis + 1, 1);
41  }
42 
43  Tensor* Y = context->Output(0, TensorShape(expanded_shape));
44  CopyCpuTensor(X, Y);
45 
46  return Status::OK();
47  }
48 };
